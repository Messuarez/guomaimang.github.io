                    Counting Prime Numbers

                    #include <iostream>
using namespace std;

bool isPrime(int n) {
  for (int i = 2; i < n; i++) {
    if (n % i == 0) {
      return false; 
    } 
  }
  return true; 
}

int countPrime(int n) {
  // base case
  if (n == 2) {
    return 1;
  }
  // general case
  if (isPrime(n)) {
    return countPrime(n - 1) + 1;
  } else{
    return countPrime(n - 1);
  } 
}

int main() {
int input;
cout << "Please enter an integer: "; 
cin >> input;
cout << "Total number of prime between 1 and " << input << " is " << countPrime(input) << "." << endl; 
}
